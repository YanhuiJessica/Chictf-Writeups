      __int64 __fastcall main(__int64 a1, char **a2, char **a3)
{
__int64 result; // rax
// 当发生缓冲区溢出时，赋给变量 v4 的值将覆盖变量 v5
char v4; // [rsp+0h] [rbp-20h]
unsigned int v5; // [rsp+8h] [rbp-18h]
// 20h - 18h = 8h
// 变量 v4 与变量 v5 的栈空间相差 8 个字节
unsigned __int64 v6; // [rsp+18h] [rbp-8h]

v6 = __readfsqword(0x28u);
setbuf(stdin, 0LL);
setbuf(stdout, 0LL);
setbuf(stderr, 0LL);
puts("What's Your Birth?");
__isoc99_scanf("%d", &v5);
while ( getchar() != 10 )
    ;
if ( v5 == 1926 ) // 无法直接通过输入使 v5 的值为 1926
{
    puts("You Cannot Born In 1926!");
    result = 0LL;
}
else
{
    puts("What's Your Name?");
    gets(&v4);  // gets 不检查输入长度
    printf("You Are Born In %d\n", v5);
    // 目标：通过缓冲区溢出使 v5 的值为 1926
    if ( v5 == 1926 )
    {
    puts("You Shall Have Flag.");
    system("cat flag");
    }
    else
    {
    puts("You Are Naive.");
    puts("You Speed One Second Here.");
    }
    result = 0LL;
}
return result;
}
    • 变量 v4 与变量 v5 的栈空间相差 8 个字节，当What's Your Name?提示输入的字符串超过 8 位时，将发生缓冲区溢出。1926 转换为十六进制是0x0786（均为不可打印字符），注意使用小端序
